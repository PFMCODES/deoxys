cmake_minimum_required(VERSION 3.25)

project(Deoxys
    VERSION 0.1.0
    LANGUAGES C CXX
)

set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# GLFW
add_subdirectory(deps/GLFW)

# GLAD
add_library(glad
    deps/glad/src/gl.c
)

target_include_directories(glad PUBLIC
    deps/glad/include
)

# Deoxys executable
add_executable(deoxys
    main.cpp
    window.cpp
    Colorion/Colorion.cpp
    Archer/Archer.cpp
    System/theme/theme.cpp
)

target_link_libraries(deoxys PRIVATE
    glfw
    glad
)
